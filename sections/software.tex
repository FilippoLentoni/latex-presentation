\section{Traditional Software Evaluation}

% ---------------- Slide 5 ----------------
\begin{frame}{1) Traditional Software Engineering}
\begin{itemize}
  \item Deterministic program: same input $\Rightarrow$ same output
  \item Correctness is (mostly) binary
  \item Failures are reproducible: bugs, regressions, integration issues
  \item Unit tests, integration tests, end-to-end tests
\end{itemize}
\end{frame}
