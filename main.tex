% ---------------- Slide 3 (Merged: motivation + thesis) ----------------
\section{Motivation \& framing}

\begin{frame}{Why evaluation gets harder (and why agents force the issue)}
\begin{itemize}
  \item Modern AI systems fail in more ways than ``wrong output''
  \item Each layer adds uncertainty and expands what must be evaluated
  \item Agents force evaluation to become \textbf{continuous} (not a pre-launch checkbox)
\end{itemize}

\vspace{0.5em}
\begin{block}{Core thesis: evaluation complexity compounds}
Uncertainty shifts over time:
\begin{center}
\textbf{Inputs} $\rightarrow$ \textbf{Learned parameters} $\rightarrow$ \textbf{Token trajectories} $\rightarrow$ \textbf{System behavior \& feedback loops}
\end{center}
Each step \textbf{builds on} the previous layer: you keep the old evaluation problems and add new ones.
\end{block}
\end{frame}

% ---------------- Slide 4 ----------------
\begin{frame}{What are we evaluating, really?}
\begin{block}{Definition}
Evaluation = measuring the gap between intended behavior and observed behavior under uncertainty.
\end{block}
\begin{itemize}
  \item Object: output, prediction, completion, trajectory, or workflow outcome
  \item Constraints: latency, cost, safety, format, coverage, adoption
  \item Evidence: tests, datasets, human labels, traces, dashboards
\end{itemize}
\end{frame}
